String processor(float td, float h, float tl) {
  String out = String(index_html);
  out.replace("%TEMP_DHT%", String(td, 1));
  out.replace("%HUM%", String(h, 0));
  out.replace("%TEMP_LM%", String(tl, 1));
  return out;
}

void setup() {
  Serial.begin(115200);

  // 1. Inisialisasi OLED
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED Failed");
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);
  display.setCursor(0,0);
  display.println("SYSTEM BOOTING...");
  display.display();

  // 2. WIFI
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  Serial.print("Connecting WiFi");
  display.println("Connecting WiFi...");
  display.display();

  int timeout = 0;
  while (WiFi.status() != WL_CONNECTED && timeout < 20) {
    delay(500);
    Serial.print(".");
    timeout++;
  }

  if(WiFi.status() == WL_CONNECTED) {
    display.println("WiFi: CONNECTED");
    Serial.print("\nIP Address: ");
    Serial.println(WiFi.localIP());
    server.begin(); // Start Web Server
  } else {
    display.println("WiFi: TIMEOUT (Offline)");
  }
  display.display();

  // 3. SD CARD
  if(!SD.begin(SD_CS)) {
    Serial.println("SD Failed");
    display.println("SD Card: FAILED");
  } else {
    Serial.println("SD OK");
    display.println("SD Card: OK");
    File file = SD.open("/log_data.csv", FILE_APPEND);
    if(file && file.size() == 0) {
      file.println("Waktu_ms,Temp_DHT,Hum_DHT,Temp_LM35");
      file.close();
    }
  }
  
  display.display();
  dht.begin();
  delay(2000);
}

void loop() {
  // --- BACA SENSOR ---
  t_dht = dht.readTemperature();
  h_dht = dht.readHumidity();

  // LM35 - Oversampling 20x
  long rawAnalog = 0;
  for(int i=0; i<20; i++) { rawAnalog += analogRead(LM35_PIN); delay(2); }
  float avgAnalog = rawAnalog / 20.0;
  t_lm35 = (avgAnalog / 4095.0) * 3300.0 / 10.0;

  // --- SAVE SD CARD ---
  File dataFile = SD.open("/log_data.csv", FILE_APPEND);
  if (dataFile) {
    dataFile.printf("%lu,%.2f,%.2f,%.2f\n", millis(), t_dht, h_dht, t_lm35);
    dataFile.close();
  }

  // --- UPDATE OLED ---
  display.clearDisplay();
  display.setCursor(0,0);
  display.println("NAJMI SMART LOG");
  display.drawLine(0, 10, 127, 10, SSD1306_WHITE);
  display.setCursor(0, 18);
  display.printf("DHT : %.1f C | %.0f%%", t_dht, h_dht);
  display.setCursor(0, 32);
  display.printf("LM35: %.1f C", t_lm35);
  
  display.setCursor(0, 54);
  if (millis() - lastCloud < 3000) {
    display.print("Cloud: SENDING...");
  } else {
    display.print(WiFi.status() == WL_CONNECTED ? "WiFi: ON" : "WiFi: OFF");
    display.print(" | SD: OK");
  }
  display.display();

  // --- SEND TO CLOUD (Setiap 15 Detik) ---
  if (millis() - lastCloud >= intervalCloud) {
    if (WiFi.status() == WL_CONNECTED) {
      HTTPClient http;
      String url = GOOGLE_URL + "?dht=" + String(t_dht) + "&ds=0" + "&lm=" + String(t_lm35) + "&rh=" + String(h_dht);
      http.begin(url.c_str());
      http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);
      int code = http.GET();
      Serial.printf("Cloud Send, Code: %d\n", code);
      http.end();
    }
    lastCloud = millis();
  }

  // --- WEB SERVER HANDLER ---
  WiFiClient client = server.available();
  if (client) {
    String currentLine = "";
    while (client.connected()) {
      if (client.available()) {
        char c = client.read();
        if (c == '\n') {
          if (currentLine.length() == 0) {
            client.println("HTTP/1.1 200 OK");
            client.println("Content-type:text/html");
            client.println("Connection: close");
            client.println();
            client.print(processor(t_dht, h_dht, t_lm35));
            break;
          } else { currentLine = ""; }
        } else if (c != '\r') { currentLine += c; }
      }
    }
    client.stop();
  }
}
